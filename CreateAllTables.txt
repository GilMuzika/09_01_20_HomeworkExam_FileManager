CREATE TABLE "Students" (
	"ID"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	"FIRST_NAME"	TEXT,
	"LAST_NAME"	TEXT,
	"ADDRESS"	TEXT,
	"AGE"	INTEGER,
	"EXTERNAL_ID"	TEXT
);

CREATE TABLE "Grades" (
	"COURSE_ID"	INTEGER NOT NULL,
	"STUDENT_ID"	INTEGER NOT NULL,
	"GRADE"	INTEGER,
	"STUDENT_EXT_ID"	TEXT,
	"COURSE_EXT_ID"	TEXT,
	FOREIGN KEY("COURSE_ID") REFERENCES "Courses"("ID"),
	FOREIGN KEY("STUDENT_ID") REFERENCES "Students"("ID"),
	PRIMARY KEY("COURSE_ID","STUDENT_ID")
);

CREATE TABLE "Info" (
	"COURSE_ID_WOTH_MAX_STUDENTS"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	"COURSE_ID_WITH_MAX_AVG"	INTEGER,
	"STUDENT_ID_MAX_ENROLL_COUNT"	INTEGER,
	"STUDENT_ID_MAX_HIGHEST_COUNT"	INTEGER,
	"STUDENT_LAST_NAME_WITH_MAX_COUNT"	INTEGER
);

CREATE TABLE "Courses" (
	"ID"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	"NAME"	TEXT,
	"HOURS"	INTEGER,
	"AVG_GRADE"	NUMERIC,
	"NUM_OF_STUDENTS"	INTEGER,
	"HIGHEST_GRADE"	INTEGER,
	"EXTERNAL_ID"	TEXT
);
