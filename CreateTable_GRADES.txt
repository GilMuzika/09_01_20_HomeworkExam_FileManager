CREATE TABLE "Grades" (
	"COURSE_ID"	INTEGER NOT NULL,
	"STUDENT_ID"	INTEGER NOT NULL,
	"GRADE"	INTEGER,
	"STUDENT_EXT_ID"	TEXT,
	"COURSE_EXT_ID"	TEXT,
	FOREIGN KEY("COURSE_ID") REFERENCES "Courses"("ID"),
	FOREIGN KEY("STUDENT_ID") REFERENCES "Students"("ID"),
	PRIMARY KEY("COURSE_ID","STUDENT_ID")
);
